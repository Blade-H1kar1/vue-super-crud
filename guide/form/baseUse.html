    # 基本使用

    # 基本使用

    v-model="data" 绑定的表单值data可以不用声明属性name,gender来绑定响应式，组件内部会自动进行响应式处理
    action 配置表单按钮

    <template>
  <div>
    <sc-form
      :options="options"
      :loading.sync="loading"
      v-model="data"
      @submit="submit"
    >
    </sc-form>
  </div>
</template>

<script>
export default {
  data() {
    return {
      loading: false,
      options: {
        action: {
          submit: {},
          reset: {},
        },
        columns: 2,
        columnGap: "10px",
        renderColumns: [
          {
            label: "姓名",
            prop: "name",
            required: true,
          },
          {
            label: "性别",
            prop: "gender",
          },
        ],
      },
      data: {},
    };
  },
  methods: {
    submit(cb, form) {
      setTimeout(() => {
        cb();
      }, 500);
    },
  },
};
</script>

    # 设置 columns 固定每行列数

    columns 表单列数，columnGap 列间距， rowGap 行间距
    renderColumns项中，widthSize 占据列数，heightSize 占据行数

    <template>
  <div>
    <sc-form :options="options" :loading.sync="loading" v-model="data">
    </sc-form>
  </div>
</template>

<script>
export default {
  data() {
    return {
      loading: false,
      options: {
        layout: "grid",
        columns: 2,
        columnGap: "10px",
        rowGap: "10px",
        renderColumns: [
          {
            label: "widthSize",
            prop: "widthSize",
            widthSize: 2,
          },
          {
            label: "heightSize",
            prop: "heightSize",
            heightSize: 2,
            comp: {
              type: "textarea",
              autosize: { minRows: 4 },
            },
          },
          {
            label: "姓名",
            prop: "name",
          },
          {
            label: "年龄",
            prop: "age",
          },
        ],
      },
      data: {},
    };
  },
};
</script>

    # 校验

    renderColumns项中，required 必填，rules 匹配规则模板和自定义规则
    详情参考crud-校验

    <template>
  <div>
    <sc-form :options="options" :loading.sync="loading" v-model="data">
    </sc-form>
  </div>
</template>

<script>
export default {
  data() {
    return {
      loading: false,
      options: {
        columns: 2,
        action: {
          submit: {},
          reset: {},
        },
        renderColumns: [
          {
            label: "必填项",
            prop: "required",
            required: true,
          },
          {
            label: "正则",
            prop: "regulars",
            rules: ["number"],
          },
          {
            label: "自定义",
            prop: "custom",
            rules: [
              {
                validator: (rule, value, callback) => {

                    callback();
                  } else {
                    callback(new Error("error1"));
                  }
                },
                trigger: "change",
              },
            ],
          },
        ],
      },
      data: {},
    };
  },
};
</script>

    # 数据格式化

    当后端接口数据格式与前端不一致时，可以使用formatData进行数据格式化
    formatData.input 将外部后端接口的数据格式转化成内部表单所需的格式
    formatData.output 将内部表单的数据格式反转回原来的格式
    formatData 接收字符串时则匹配预设的全局格式化模板
    formatData.getFormatValue = true 可获取格式化后的值，默认绑定在 scope.row[$ + prop] 上，如果配置getFormatValue为字符串则绑定在 scope.row[getFormatValue] 上

    输入值：str2array-1,2
    <template>
  <div>
    <div style="margin: 0 0 20px 40px;">
      <div>输入值：str2array-{{ data.strToArr }}</div>
    </div>
    <sc-form
      ref="form"
      :options="options"
      :loading.sync="loading"
      v-model="data"
      @submit="submit"
    >
    </sc-form>
  </div>
</template>

<script>
export default {
  data() {
    return {
      loading: false,
      dictData: [
        {
          value: "1",
          label: "张三",
        },
        {
          value: "2",
          label: "李四",
        },
      ],

      data: {
        strToArr: "1,2",
      },
    };
  },
  computed: {
    options() {
      return {
        action: {
          submit: {},
          reset: {},
        },
        columns: 2,
        columnGap: "10px",
        renderColumns: [
          {
            label: "字符转数组",
            prop: "strToArr",
            comp: {
              name: "sc-checkbox",
              options: this.dictData,
            },
            formatData: {
              formatValue: true,
              input: (value, { row }) => {
                if (!value) {
                  return [];
                }
                return typeof value === "string" ? value.split(",") : value;
              },
              output: (value, { row }) => {
                if (!value) {
                  return "";
                }
                return value.join(",");
              },
            },
          },
          {
            label: "字符转数组-快捷方式",
            prop: "strToArr",
            comp: {
              name: "sc-checkbox",
              options: this.dictData,
            },
            formatData: "strToArr",
          },
        ],
      };
    },
  },
  methods: {
    submit(cb, form) {
      setTimeout(() => {
        cb();
      }, 500);
    },
  },
};
</script>

    # 深度绑定

    deepProp 绑定的对象数据是双向的，改变任意一个，另外一个也会改变

    改变deep值 改变name值 { "deep": { "deep": { "name": "deep" } } } 
    <template>
  <div>
    <el-button
      type="primary"
      size="small"
      @click="data.deep.deep.name = '改变deep值'"
      >改变deep值</el-button
    >
    <el-button type="primary" size="small" @click="data.name = '改变name值'"
      >改变name值</el-button
    >
    {{ data }}
    <sc-form :options="options" v-model="data"> </sc-form>
  </div>
</template>

<script>
export default {
  data() {
    return {
      dictData: [
        {
          value: "1",
          label: "张三",
        },
        {
          value: "2",
          label: "李四",
        },
      ],

      data: {
        deep: {
          deep: {
            name: "deep",
          },
        },
      },
    };
  },
  computed: {
    options() {
      return {
        columns: 2,
        columnGap: "10px",
        renderColumns: [
          {
            label: "姓名",
            prop: "name",
            deepProp: "deep.deep.name",
          },
        ],
      };
    },
  },
};
</script>
